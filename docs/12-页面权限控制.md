# 12 - 页面权限控制（菜单 + 按钮动态渲染）

## 📖 本课目标

- 实现"获取当前用户权限"API
- 侧边栏菜单根据权限动态显示
- 按钮级别的权限控制
- 未授权访问重定向

---

## 🧠 核心思路

```
用户登录 → JWT 含 userId
    ↓
前端请求 /api/auth/me → 返回用户信息 + 权限列表
    ↓
权限列表存入 Context（全局状态）
    ↓
侧边栏：只渲染有 menu 权限的菜单项
按钮：有对应 button 权限才显示操作按钮
```

## 第一步：当前用户信息 API

新建 `src/app/api/auth/me/route.ts`：

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/prisma'

// GET /api/auth/me — 获取当前登录用户信息 + 权限列表
export async function GET(req: NextRequest) {
  const userId = req.headers.get('x-user-id')

  if (!userId) {
    return NextResponse.json({ error: '未登录' }, { status: 401 })
  }

  const user = await prisma.user.findUnique({
    where: { id: userId },
    select: {
      id: true,
      email: true,
      name: true,
      avatar: true,
      status: true,
      department: { select: { id: true, name: true } },
      userRoles: {
        select: {
          role: {
            select: {
              id: true,
              name: true,
              rolePermissions: {
                select: {
                  permission: {
                    select: { id: true, name: true, type: true, path: true, icon: true, module: true }
                  }
                }
              }
            }
          }
        }
      }
    }
  })

  if (!user) {
    return NextResponse.json({ error: '用户不存在' }, { status: 404 })
  }

  // 展开权限列表（去重）
  const permissionSet = new Map<string, any>()
  for (const userRole of user.userRoles) {
    for (const rp of userRole.role.rolePermissions) {
      permissionSet.set(rp.permission.id, rp.permission)
    }
  }

  const permissions = Array.from(permissionSet.values())
  const permissionNames = permissions.map(p => p.name)  // 用于快速判断

  return NextResponse.json({
    data: {
      id: user.id,
      email: user.email,
      name: user.name,
      avatar: user.avatar,
      roles: user.userRoles.map(ur => ur.role.name),
      permissions,
      permissionNames,
    }
  })
}
```

## 第二步：权限 Context（全局状态）

新建 `src/contexts/auth-context.tsx`：

```tsx
'use client'

import { createContext, useContext, useEffect, useState, ReactNode } from 'react'

interface Permission {
  id: string
  name: string
  type: string
  path: string | null
  icon: string | null
  module: string
}

interface AuthUser {
  id: string
  email: string | null
  name: string | null
  avatar: string | null
  roles: string[]
  permissions: Permission[]
  permissionNames: string[]
}

interface AuthContextType {
  user: AuthUser | null
  loading: boolean
  hasPermission: (name: string) => boolean
  hasAnyPermission: (names: string[]) => boolean
  refetch: () => void
}

const AuthContext = createContext<AuthContextType>({
  user: null,
  loading: true,
  hasPermission: () => false,
  hasAnyPermission: () => false,
  refetch: () => {},
})

export function AuthProvider({ children }: { children: ReactNode }) {
  const [user, setUser] = useState<AuthUser | null>(null)
  const [loading, setLoading] = useState(true)

  async function fetchUser() {
    try {
      const res = await fetch('/api/auth/me')
      if (res.ok) {
        const data = await res.json()
        setUser(data.data)
      } else {
        setUser(null)
      }
    } catch {
      setUser(null)
    } finally {
      setLoading(false)
    }
  }

  useEffect(() => { fetchUser() }, [])

  // 判断是否有某个权限
  function hasPermission(name: string): boolean {
    if (!user) return false
    // admin 角色拥有所有权限
    if (user.roles.includes('admin')) return true
    return user.permissionNames.includes(name)
  }

  // 判断是否有任意一个权限
  function hasAnyPermission(names: string[]): boolean {
    return names.some(name => hasPermission(name))
  }

  return (
    <AuthContext.Provider value={{ user, loading, hasPermission, hasAnyPermission, refetch: fetchUser }}>
      {children}
    </AuthContext.Provider>
  )
}

// 自定义 hook，方便使用
export const useAuth = () => useContext(AuthContext)
```

## 第四步：侧边栏动态菜单

修改 `src/components/app-sidebar.tsx`，根据权限动态过滤菜单：

```tsx
'use client'

import {
  Sidebar, SidebarContent, SidebarGroup, SidebarGroupLabel,
  SidebarGroupContent, SidebarMenu, SidebarMenuItem, SidebarMenuButton,
  SidebarHeader, SidebarFooter,
} from '@/components/ui/sidebar'
import { Avatar, AvatarFallback, AvatarImage } from '@/components/ui/avatar'
import { LayoutDashboard, Users, Shield, Building, FolderKanban, LogOut } from 'lucide-react'
import Link from 'next/link'
import { useAuth } from '@/contexts/auth-context'
import { usePathname } from 'next/navigation'

// 每个菜单项对应的权限名
const menuItems = [
  { title: '仪表盘', url: '/dashboard', icon: LayoutDashboard, permission: 'menu:dashboard' },
  { title: '用户管理', url: '/users', icon: Users, permission: 'menu:user' },
  { title: '角色管理', url: '/roles', icon: Shield, permission: 'menu:role' },
  { title: '部门管理', url: '/departments', icon: Building, permission: 'menu:dept' },
  { title: '项目管理', url: '/projects', icon: FolderKanban, permission: 'menu:project' },
]

export function AppSidebar() {
  const { user, hasPermission } = useAuth()
  const pathname = usePathname()

  // 只显示有权限的菜单
  const visibleMenus = menuItems.filter(item => hasPermission(item.permission))

  return (
    <Sidebar>
      <SidebarHeader className="p-4">
        <div className="flex items-center gap-2">
          <Shield className="h-6 w-6 text-primary" />
          <span className="font-semibold text-lg">RBAC 系统</span>
        </div>
      </SidebarHeader>

      <SidebarContent>
        <SidebarGroup>
          <SidebarGroupLabel>功能菜单</SidebarGroupLabel>
          <SidebarGroupContent>
            <SidebarMenu>
              {visibleMenus.map((item) => (
                <SidebarMenuItem key={item.title}>
                  <SidebarMenuButton asChild isActive={pathname === item.url}>
                    <Link href={item.url}>
                      <item.icon />
                      <span>{item.title}</span>
                    </Link>
                  </SidebarMenuButton>
                </SidebarMenuItem>
              ))}
            </SidebarMenu>
          </SidebarGroupContent>
        </SidebarGroup>
      </SidebarContent>

      <SidebarFooter className="p-4">
        {/* 当前用户信息 */}
        {user && (
          <div className="flex items-center gap-2 mb-3 px-2">
            <Avatar className="h-8 w-8">
              <AvatarImage src={user.avatar ?? undefined} />
              <AvatarFallback>{user.name?.[0]?.toUpperCase()}</AvatarFallback>

            <div className="flex-1 min-w-0">
              <p className="text-sm font-medium truncate">{user.name}</p>

            </div>
          </div>
        )}
        <SidebarMenu>
          <SidebarMenuItem>

              <Link href="/api/auth/logout">
                <LogOut />
                <span>退出登录</span>
              </Link>
            </SidebarMenuButton>
          </SidebarMenuItem>
        </SidebarMenu>
      </SidebarFooter>
```

## 第五步：按钮权限控制组件

新建 `src/components/permission-guard.tsx`：

```tsx
'use client'

import { useAuth } from '@/contexts/auth-context'
import { ReactNode } from 'react'

interface PermissionGuardProps {
  permission: string    // 需要的权限名
  children: ReactNode   // 有权限时显示的内容
  fallback?: ReactNode  // 无权限时显示的内容（默认不显示）
}

export function PermissionGuard({ permission, children, fallback = null }: PermissionGuardProps) {
  const { hasPermission } = useAuth()

  if (!hasPermission(permission)) return <>{fallback}</>
  return <>{children}</>
}
```

```tsx
import { PermissionGuard } from '@/components/permission-guard'

// 新增按钮 — 需要 user:create 权限
<PermissionGuard permission="user:create">
  <Button onClick={() => { setEditUser(null); setDialogOpen(true) }}>
    <Plus className="mr-2 h-4 w-4" />
    新增用户
  </Button>
</PermissionGuard>

// 编辑按钮 — 需要 user:update 权限
<PermissionGuard permission="user:update">
  <Button variant="ghost" size="icon" onClick={() => { setEditUser(user); setDialogOpen(true) }}>
    <Pencil className="h-4 w-4" />
  </Button>
</PermissionGuard>

// 删除按钮 — 需要 user:delete 权限
<PermissionGuard permission="user:delete">
  <Button
    variant="ghost" size="icon"
    className="text-destructive hover:text-destructive"
    onClick={() => handleDelete(user.id)}
  >
    <Trash2 className="h-4 w-4" />
  </Button>
</PermissionGuard>
```

## ⚠️ 注意：前端权限控制 ≠ 安全

前端隐藏按钮是 **UX 优化**，不是安全保障。攻击者可以直接调用 API。

真正的安全需要在 **API 层** 也做权限校验：

```typescript
// 在需要权限的 API 里加校验
export async function DELETE(req: NextRequest, ...) {
  const userId = req.headers.get('x-user-id')!

  // 查询用户权限
  const userPerms = await getUserPermissions(userId)

  if (!userPerms.includes('user:delete')) {
    return NextResponse.json({ error: '无权限' }, { status: 403 })
  }

  // 继续删除逻辑...
}
```
