## 🧠 核心概念：为什么在 useEffect 里 fetch？

```tsx
// ❌ 错误：直接在组件函数体里 fetch
export default function UsersPage() {
  const data = await fetch('/api/users')  // 不能这样写！
  // 函数组件不是 async 函数，每次渲染都会重新 fetch
}

// ✅ 正确：在 useEffect 里 fetch
export default function UsersPage() {
  const [users, setUsers] = useState([])

  useEffect(() => {
    fetch('/api/users')
      .then(r => r.json())
      .then(d => setUsers(d.data))
  }, [])  // [] = 只在组件挂载时执行一次
}
```

**useEffect 依赖数组说明：**
- `[]` — 只执行一次（组件挂载）
- `[page, keyword]` — page 或 keyword 变化时重新执行
- 不传 — 每次渲染都执行（危险！会死循环）

## 用户管理页面

`src/app/(dashboard)/users/page.tsx`

```tsx
'use client'

import { useEffect, useState } from 'react'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Badge } from '@/components/ui/badge'
import { Avatar, AvatarFallback, AvatarImage } from '@/components/ui/avatar'
import {
  Table, TableBody, TableCell, TableHead,
  TableHeader, TableRow,
} from '@/components/ui/table'
import { Card, CardContent, CardHeader } from '@/components/ui/card'
import { Plus, Search, Pencil, Trash2, Loader2 } from 'lucide-react'
import { UserDialog } from '@/components/users/user-dialog'

interface User {
  id: string
  email: string
  name: string
  avatar: string | null
  status: boolean
  createdAt: string
  department: { id: string; name: string } | null
  userRoles: { role: { id: string; name: string } }[]
}

export default function UsersPage() {
  const [users, setUsers] = useState<User[]>([])
  const [total, setTotal] = useState(0)
  const [loading, setLoading] = useState(true)
  const [keyword, setKeyword] = useState('')
  const [page, setPage] = useState(1)
  const [dialogOpen, setDialogOpen] = useState(false)
  const [editUser, setEditUser] = useState<User | null>(null)

  async function fetchUsers() {
    setLoading(true)
    try {
      const params = new URLSearchParams({
        page: String(page),
        pageSize: '10',
        ...(keyword && { keyword }),
      })
      const res = await fetch(`/api/users?${params}`)
      const data = await res.json()
      setUsers(data.data)
      setTotal(data.total)
    } finally {
      setLoading(false)
    }
  }

  // page 或 keyword 变化时重新获取数据
  useEffect(() => { fetchUsers() }, [page, keyword])

  // 删除用户
  async function handleDelete(id: string) {
    if (!confirm('确认删除该用户？')) return
    await fetch(`/api/users/${id}`, { method: 'DELETE' })
    fetchUsers()
  }

  // 点击状态 Badge 直接切换（不需要弹窗，简洁高效）
  async function handleToggleStatus(user: User) {
    await fetch(`/api/users/${user.id}`, {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ status: !user.status }),
    })
    fetchUsers()
  }

  return (
    <div>
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-bold">用户管理</h1>
        <Button onClick={() => { setEditUser(null); setDialogOpen(true) }}>
          <Plus className="mr-2 h-4 w-4" />
          新增用户
        </Button>
      </div>

      <Card>
        <CardHeader>
          <div className="flex items-center gap-2">
            <div className="relative flex-1 max-w-sm">
              <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
              <Input
                placeholder="搜索用户名/邮箱..."
                className="pl-9"
                value={keyword}
                onChange={e => { setKeyword(e.target.value); setPage(1) }}
              />
            </div>
            <span className="text-sm text-muted-foreground">共 {total} 条</span>
          </div>
        </CardHeader>

        <CardContent>
          {loading ? (
            <div className="flex justify-center py-12">
              <Loader2 className="h-6 w-6 animate-spin text-muted-foreground" />
            </div>
          ) : (
            <Table>
              <TableHeader>
                <TableRow>
                  <TableHead>用户</TableHead>
                  <TableHead>邮箱</TableHead>
                  <TableHead>角色</TableHead>
                  <TableHead>部门</TableHead>
                  <TableHead>状态</TableHead>
                  <TableHead>创建时间</TableHead>
                  <TableHead className="text-right">操作</TableHead>
                </TableRow>
              </TableHeader>
              <TableBody>
                {users.map(user => (
                  <TableRow key={user.id}>
                    <TableCell>
                      <div className="flex items-center gap-2">
                        <Avatar className="h-8 w-8">
                          <AvatarImage src={user.avatar ?? undefined} />
                          {/* AvatarFallback：头像加载失败时显示姓名首字母 */}
                          <AvatarFallback>
                            {user.name?.[0]?.toUpperCase() ?? '?'}
                          </AvatarFallback>
                        </Avatar>
                        <span className="font-medium">{user.name}</span>
                      </div>
                    </TableCell>
                    <TableCell className="text-muted-foreground">{user.email}</TableCell>
                    <TableCell>
                      <div className="flex gap-1 flex-wrap">
                        {user.userRoles.length > 0
                          ? user.userRoles.map(ur => (
                              <Badge key={ur.role.id} variant="secondary">
                                {ur.role.name}
                              </Badge>
                            ))
                          : <span className="text-muted-foreground text-sm">—</span>
                        }
                      </div>
                    </TableCell>
                    <TableCell>{user.department?.name ?? '—'}</TableCell>
                    <TableCell>
                      {/* 点击 Badge 直接切换状态，无需弹窗 */}
                      <Badge
                        variant={user.status ? 'default' : 'destructive'}
                        className="cursor-pointer"
                        onClick={() => handleToggleStatus(user)}
                      >
                        {user.status ? '正常' : '禁用'}
                      </Badge>
                    </TableCell>
                    <TableCell className="text-muted-foreground text-sm">
                      {new Date(user.createdAt).toLocaleDateString('zh-CN')}
                    </TableCell>
                    <TableCell className="text-right">
                      <div className="flex justify-end gap-2">
                        <Button
                          variant="ghost" size="icon"
                          onClick={() => { setEditUser(user); setDialogOpen(true) }}
                        >
                          <Pencil className="h-4 w-4" />
                        </Button>
                        <Button
                          variant="ghost" size="icon"
                          className="text-destructive hover:text-destructive"
                          onClick={() => handleDelete(user.id)}
                        >
                          <Trash2 className="h-4 w-4" />
                        </Button>
                      </div>
                    </TableCell>
                  </TableRow>
                ))}
              </TableBody>
            </Table>
          )}

          {/* 分页 */}
          {total > 10 && (
            <div className="flex justify-end gap-2 mt-4">
              <Button
                variant="outline" size="sm"
                disabled={page === 1}
                onClick={() => setPage(p => p - 1)}
              >上一页</Button>
              <Button
                variant="outline" size="sm"
                disabled={page * 10 >= total}
                onClick={() => setPage(p => p + 1)}
              >下一页</Button>
            </div>
          )}
        </CardContent>
      </Card>

      {/* 新增/编辑弹窗 */}
      <UserDialog
        open={dialogOpen}
        user={editUser}
        onClose={() => setDialogOpen(false)}
        onSuccess={fetchUsers}
      />
    </div>
  )
}
```

## 新增/编辑弹窗组件

`src/components/users/user-dialog.tsx`

**核心设计：新增和编辑共用一个组件**

```tsx
// 通过 user prop 区分模式：
// user = null  → 新增模式（显示邮箱+密码字段）
// user = {...} → 编辑模式（邮箱不可改，密码不显示）
const isEdit = !!user
```

```tsx
'use client'

import { useEffect, useState } from 'react'
import {
  Dialog, DialogContent, DialogHeader,
  DialogTitle, DialogFooter,
} from '@/components/ui/dialog'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'
import { Checkbox } from '@/components/ui/checkbox'
import { Loader2 } from 'lucide-react'

interface Role {
  id: string
  name: string
  description: string
}

interface User {
  id: string
  email: string
  name: string
  status: boolean
  userRoles: { role: { id: string; name: string } }[]
}

interface UserDialogProps {
  open: boolean
  user?: User | null
  onClose: () => void
  onSuccess: () => void
}

export function UserDialog({ open, user, onClose, onSuccess }: UserDialogProps) {
  const isEdit = !!user
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState('')
  const [roles, setRoles] = useState<Role[]>([])
  const [form, setForm] = useState({
    email: '',
    password: '',
    name: '',
    roleIds: [] as string[],
  })

  // 加载角色列表（弹窗打开时）
  useEffect(() => {
    fetch('/api/roles')
      .then(r => r.json())
      .then(d => setRoles(d.data))
  }, [])

  // 编辑时回填数据，新增时清空
  useEffect(() => {
    if (user) {
      setForm({
        email: user.email,
        password: '',
        name: user.name,
        roleIds: user.userRoles.map(ur => ur.role.id),
      })
    } else {
      setForm({ email: '', password: '', name: '', roleIds: [] })
    }
    setError('')
  }, [user, open])  // user 或 open 变化时触发

  function toggleRole(roleId: string) {
    setForm(f => ({
      ...f,
      // 已选中 → 取消；未选中 → 添加
      roleIds: f.roleIds.includes(roleId)
        ? f.roleIds.filter(id => id !== roleId)
        : [...f.roleIds, roleId],
    }))
  }

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    setLoading(true)
    setError('')

    try {
      const url = isEdit ? `/api/users/${user.id}` : '/api/users'
      const method = isEdit ? 'PATCH' : 'POST'
      // 编辑时不传 email 和 password（不允许修改）
      const body = isEdit
        ? { name: form.name, roleIds: form.roleIds }
        : { email: form.email, password: form.password, name: form.name, roleIds: form.roleIds }

      const res = await fetch(url, {
        method,
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body),
      })

      const data = await res.json()
      if (!res.ok) {
        setError(data.error)
        return
      }

      onSuccess()  // 刷新列表
      onClose()    // 关闭弹窗
    } catch {
      setError('网络错误')
    } finally {
      setLoading(false)
    }
  }

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent className="max-w-md">
        <DialogHeader>
          <DialogTitle>{isEdit ? '编辑用户' : '新增用户'}</DialogTitle>
        </DialogHeader>

        <form onSubmit={handleSubmit} className="space-y-4">
          {/* 新增时才显示邮箱 */}
          {!isEdit && (
            <div className="space-y-2">
              <Label>邮箱 *</Label>
              <Input
                type="email"
                placeholder="user@example.com"
                value={form.email}
                onChange={e => setForm(f => ({ ...f, email: e.target.value }))}
                required
              />
            </div>
          )}

          <div className="space-y-2">
            <Label>姓名</Label>
            <Input
              placeholder="用户姓名"
              value={form.name}
              onChange={e => setForm(f => ({ ...f, name: e.target.value }))}
            />
          </div>

          {/* 新增时才显示密码 */}
          {!isEdit && (
            <div className="space-y-2">
              <Label>密码 *</Label>
              <Input
                type="password"
                placeholder="至少 6 位"
                value={form.password}
                onChange={e => setForm(f => ({ ...f, password: e.target.value }))}
                required
                minLength={6}
              />
            </div>
          )}

          <div className="space-y-2">
            <Label>分配角色</Label>
            <div className="border rounded-md p-3 space-y-2">
              {roles.map(role => (
                <div key={role.id} className="flex items-center gap-2">
                  <Checkbox
                    id={role.id}
                    checked={form.roleIds.includes(role.id)}
                    onCheckedChange={() => toggleRole(role.id)}
                  />
                  <label htmlFor={role.id} className="text-sm cursor-pointer">
                    <span className="font-medium">{role.name}</span>
                    {role.description && (
                      <span className="text-muted-foreground ml-1">— {role.description}</span>
                    )}
                  </label>
                </div>
              ))}
            </div>
          </div>

          {error && <p className="text-sm text-destructive">{error}</p>}

          <DialogFooter>
            <Button type="button" variant="outline" onClick={onClose}>取消</Button>
            <Button type="submit" disabled={loading}>
              {loading && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
              {isEdit ? '保存' : '创建'}
            </Button>
          </DialogFooter>
        </form>
      </DialogContent>
    </Dialog>
  )
}
```

function flatten(items: any[]): any[] {

### 完整页面代码

```tsx
'use client'

import { useEffect, useState } from 'react'
import { Button } from '@/components/ui/button'
import { Badge } from '@/components/ui/badge'
import {
  Table, TableBody, TableCell, TableHead,
  TableHeader, TableRow,
} from '@/components/ui/table'
import { Card, CardContent, CardHeader } from '@/components/ui/card'
import {
  Dialog, DialogContent, DialogHeader,
  DialogTitle, DialogFooter,
} from '@/components/ui/dialog'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'
import { Checkbox } from '@/components/ui/checkbox'
import { Plus, Pencil, Trash2, Loader2 } from 'lucide-react'

interface Permission {
  id: string
  name: string
  description: string
  type: string
  module: string
  children?: Permission[]
}

interface Role {
  id: string
  name: string
  description: string
  createdAt: string
  _count: { userRoles: number }
  rolePermissions: { permission: { id: string; name: string; type: string } }[]
}

export default function RolesPage() {
  const [roles, setRoles] = useState<Role[]>([])
  const [permissions, setPermissions] = useState<Permission[]>([])
  const [loading, setLoading] = useState(true)
  const [dialogOpen, setDialogOpen] = useState(false)
  const [editRole, setEditRole] = useState<Role | null>(null)
  const [saving, setSaving] = useState(false)
  const [error, setError] = useState('')
  const [form, setForm] = useState({
    name: '', description: '', permissionIds: [] as string[]
  })

  async function fetchRoles() {
    setLoading(true)
    const res = await fetch('/api/roles')
    const data = await res.json()
    setRoles(data.data)
    setLoading(false)
  }

  useEffect(() => {
    fetchRoles()
    // 获取权限列表并展开树形结构
    fetch('/api/permissions')
      .then(r => r.json())
      .then(d => {
        function flatten(items: any[]): any[] {
          return items.flatMap((item: any) => [item, ...flatten(item.children || [])])
        }
        setPermissions(flatten(d.data))
      })
  }, [])

  // 弹窗打开/关闭时重置表单
  useEffect(() => {
    if (editRole) {
      setForm({
        name: editRole.name,
        description: editRole.description ?? '',
        permissionIds: editRole.rolePermissions.map(rp => rp.permission.id),
      })
    } else {
      setForm({ name: '', description: '', permissionIds: [] })
    }
    setError('')
  }, [editRole, dialogOpen])

  function togglePermission(id: string) {
    setForm(f => ({
      ...f,
      permissionIds: f.permissionIds.includes(id)
        ? f.permissionIds.filter(p => p !== id)
        : [...f.permissionIds, id],
    }))
  }

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    setSaving(true)
    setError('')
    try {
      const url = editRole ? `/api/roles/${editRole.id}` : '/api/roles'
      const method = editRole ? 'PATCH' : 'POST'
      const res = await fetch(url, {
        method,
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(form),
      })
      const data = await res.json()
      if (!res.ok) { setError(data.error); return }
      fetchRoles()
      setDialogOpen(false)
    } catch {
      setError('网络错误')
    } finally {
      setSaving(false)
    }
  }

  async function handleDelete(id: string) {
    if (!confirm('确认删除该角色？')) return
    await fetch(`/api/roles/${id}`, { method: 'DELETE' })
    fetchRoles()
  }

  // 按 module 分组权限（用于弹窗展示）
  const permByModule = permissions.reduce((acc, p) => {
    if (!acc[p.module]) acc[p.module] = []
    acc[p.module].push(p)
    return acc
  }, {} as Record<string, Permission[]>)

  return (
    <div>
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-bold">角色管理</h1>
        <Button onClick={() => { setEditRole(null); setDialogOpen(true) }}>
          <Plus className="mr-2 h-4 w-4" />
          新增角色
        </Button>
      </div>

      <Card>
        <CardHeader />
        <CardContent>
          {loading ? (
            <div className="flex justify-center py-12">
              <Loader2 className="h-6 w-6 animate-spin text-muted-foreground" />
            </div>
          ) : (
            <Table>
              <TableHeader>
                <TableRow>
                  <TableHead>角色名</TableHead>
                  <TableHead>描述</TableHead>
                  <TableHead>用户数</TableHead>
                  <TableHead>权限（预览）</TableHead>
                  <TableHead>创建时间</TableHead>
                  <TableHead className="text-right">操作</TableHead>
                </TableRow>
              </TableHeader>
              <TableBody>
                {roles.map(role => (
                  <TableRow key={role.id}>
                    <TableCell className="font-medium">{role.name}</TableCell>
                    <TableCell className="text-muted-foreground">{role.description}</TableCell>
                    <TableCell>{role._count.userRoles} 人</TableCell>
                    <TableCell>
                      <div className="flex gap-1 flex-wrap max-w-xs">
                        {/* 只显示前 3 个权限，其余用 +N 表示 */}
                        {role.rolePermissions.slice(0, 3).map(rp => (
                          <Badge key={rp.permission.id} variant="outline" className="text-xs">
                            {rp.permission.name}
                          </Badge>
                        ))}
                        {role.rolePermissions.length > 3 && (
                          <Badge variant="outline" className="text-xs">
                            +{role.rolePermissions.length - 3}
                          </Badge>
                        )}
                      </div>
                    </TableCell>
                    <TableCell className="text-muted-foreground text-sm">
                      {new Date(role.createdAt).toLocaleDateString('zh-CN')}
                    </TableCell>
                    <TableCell className="text-right">
                      <div className="flex justify-end gap-2">
                        <Button
                          variant="ghost" size="icon"
                          onClick={() => { setEditRole(role); setDialogOpen(true) }}
                        >
                          <Pencil className="h-4 w-4" />
                        </Button>
                        <Button
                          variant="ghost" size="icon"
                          className="text-destructive hover:text-destructive"
                          onClick={() => handleDelete(role.id)}
                        >
                          <Trash2 className="h-4 w-4" />
                        </Button>
                      </div>
                    </TableCell>
                  </TableRow>
                ))}
              </TableBody>
            </Table>
          )}
        </CardContent>
      </Card>

      {/* 新增/编辑弹窗 */}
      <Dialog open={dialogOpen} onOpenChange={setDialogOpen}>
        <DialogContent className="max-w-lg max-h-[80vh] overflow-y-auto">
          <DialogHeader>
            <DialogTitle>{editRole ? '编辑角色' : '新增角色'}</DialogTitle>
          </DialogHeader>
          <form onSubmit={handleSubmit} className="space-y-4">
            <div className="space-y-2">
              <Label>角色名 *</Label>
              <Input
                placeholder="如：editor"
                value={form.name}
                onChange={e => setForm(f => ({ ...f, name: e.target.value }))}
                required
                disabled={!!editRole}  // 编辑时不允许改角色名（唯一键约束）
              />
            </div>
            <div className="space-y-2">
              <Label>描述</Label>
              <Input
                placeholder="角色描述"
                value={form.description}
                onChange={e => setForm(f => ({ ...f, description: e.target.value }))}
              />
            </div>
            <div className="space-y-2">
              <Label>分配权限</Label>
              <div className="border rounded-md p-3 space-y-3">
                {Object.entries(permByModule).map(([module, perms]) => (
                  <div key={module}>
                    <p className="text-xs font-semibold text-muted-foreground uppercase mb-2">
                      {module}
                    </p>
                    <div className="space-y-1 pl-2">
                      {perms.map(p => (
                        <div key={p.id} className="flex items-center gap-2">
                          <Checkbox
                            id={p.id}
                            checked={form.permissionIds.includes(p.id)}
                            onCheckedChange={() => togglePermission(p.id)}
                          />
                          <label htmlFor={p.id} className="text-sm cursor-pointer">
                            {p.description}
                            <span className="text-muted-foreground ml-1 text-xs">({p.name})</span>
                          </label>
                        </div>
                      ))}
                    </div>
                  </div>
                ))}
              </div>
            </div>
            {error && <p className="text-sm text-destructive">{error}</p>}
            <DialogFooter>
              <Button type="button" variant="outline" onClick={() => setDialogOpen(false)}>取消</Button>
              <Button type="submit" disabled={saving}>
                {saving && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
                {editRole ? '保存' : '创建'}
              </Button>
            </DialogFooter>
          </form>
        </DialogContent>
      </Dialog>
    </div>
  )
}
```

## ⚠️ 避坑指南

### 1. 编辑时角色名不可改
角色名有 `@unique` 唯一约束，改名可能冲突，且角色名通常作为权限判断的 key，不应随意修改。

### 2. useEffect 依赖项漏写
如果 useEffect 里用了外部变量但没加到依赖数组，会出现闭包问题（拿到旧值）。ESLint 的 `exhaustive-deps` 规则会提示。

### 3. 搜索时重置页码
搜索条件变化时，必须把 page 重置为 1，否则可能出现"第 3 页但总共只有 5 条"的情况。

```tsx
onChange={e => { setKeyword(e.target.value); setPage(1) }}  // ← 别忘了 setPage(1)
```

### 4. Dialog 的 onOpenChange
Dialog 关闭时会触发 `onOpenChange(false)`，等同于调用 `onClose`，要确保关闭逻辑一致。
