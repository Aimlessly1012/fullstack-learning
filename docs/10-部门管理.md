## 第三步：部门管理 API

### `src/app/api/departments/route.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/prisma'

// GET /api/departments — 部门树形列表
export async function GET() {
  const departments = await prisma.department.findMany({
    orderBy: [{ sort: 'asc' }, { createdAt: 'asc' }],
    select: {
      id: true,
      name: true,
      description: true,
      parentId: true,
      sort: true,
      createdAt: true,
      _count: { select: { users: true } },
    },
  })

  // 构建树形结构
  return NextResponse.json({ data: buildTree(departments) })
}

function buildTree(items: any[], parentId: string | null = null): any[] {
  return items
    .filter(item => item.parentId === parentId)
    .map(item => ({
      ...item,
      children: buildTree(items, item.id),
    }))
}

// POST /api/departments — 创建部门
export async function POST(req: NextRequest) {
  try {
    const { name, description, parentId, sort } = await req.json()

    if (!name) {
      return NextResponse.json({ error: '部门名称必填' }, { status: 400 })
    }

    const dept = await prisma.department.create({
      data: { name, description, parentId: parentId || null, sort: sort ?? 0 },
      select: { id: true, name: true, description: true, parentId: true, sort: true, createdAt: true },
    })

    return NextResponse.json({ data: dept }, { status: 201 })
  } catch (error) {
    return NextResponse.json({ error: '服务器错误' }, { status: 500 })
  }
}
```

### `src/app/api/departments/[id]/route.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/prisma'

// PATCH /api/departments/:id — 更新部门
export async function PATCH(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id } = await params
    const { name, description, parentId, sort } = await req.json()

    const dept = await prisma.department.update({
      where: { id },
      data: {
        ...(name !== undefined && { name }),
        ...(description !== undefined && { description }),
        ...(parentId !== undefined && { parentId: parentId || null }),
        ...(sort !== undefined && { sort }),
      },
      select: { id: true, name: true, description: true, parentId: true, sort: true, updatedAt: true },
    })

    return NextResponse.json({ data: dept })
  } catch {
    return NextResponse.json({ error: '服务器错误' }, { status: 500 })
  }
}

// DELETE /api/departments/:id — 删除部门
export async function DELETE(
  _req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id } = await params

    // 检查是否有子部门
    const children = await prisma.department.count({ where: { parentId: id } })
    if (children > 0) {
      return NextResponse.json({ error: '请先删除子部门' }, { status: 400 })
    }

    // 检查是否有关联用户
    const users = await prisma.user.count({ where: { departmentId: id } })
    if (users > 0) {
      return NextResponse.json({ error: '该部门下有用户，无法删除' }, { status: 400 })
    }

    await prisma.department.delete({ where: { id } })
    return NextResponse.json({ message: '删除成功' })
  } catch {
    return NextResponse.json({ error: '服务器错误' }, { status: 500 })
  }
}
```

## 第四步：部门管理页面

新建 `src/app/(dashboard)/departments/page.tsx`：

```tsx
'use client'

import { useEffect, useState } from 'react'
import { Button } from '@/components/ui/button'
import { Badge } from '@/components/ui/badge'
import { Card, CardContent, CardHeader } from '@/components/ui/card'
import {
  Dialog, DialogContent, DialogHeader,
  DialogTitle, DialogFooter,
} from '@/components/ui/dialog'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'
import { Plus, Pencil, Trash2, Loader2, ChevronRight, ChevronDown, Building } from 'lucide-react'

interface Department {
  id: string
  name: string
  description: string
  parentId: string | null
  sort: number
  createdAt: string
  _count: { users: number }
  children: Department[]
}

export default function DepartmentsPage() {
  const [depts, setDepts] = useState<Department[]>([])
  const [allDepts, setAllDepts] = useState<Department[]>([])
  const [loading, setLoading] = useState(true)
  const [expanded, setExpanded] = useState<Set<string>>(new Set())
  const [dialogOpen, setDialogOpen] = useState(false)
  const [editDept, setEditDept] = useState<Department | null>(null)
  const [saving, setSaving] = useState(false)
  const [error, setError] = useState('')
  const [form, setForm] = useState({ name: '', description: '', parentId: '', sort: '0' })

  async function fetchDepts() {
    setLoading(true)
    const res = await fetch('/api/departments')
    const data = await res.json()
    setDepts(data.data)
    // 展开为平铺列表，用于父部门选择
    setAllDepts(flatten(data.data))
    setLoading(false)
  }

  function flatten(items: Department[]): Department[] {
    return items.flatMap(item => [item, ...flatten(item.children || [])])
  }

  useEffect(() => { fetchDepts() }, [])

  useEffect(() => {
    if (editDept) {
      setForm({
        name: editDept.name,
        description: editDept.description ?? '',
        parentId: editDept.parentId ?? '',
        sort: String(editDept.sort),
      })
    } else {
      setForm({ name: '', description: '', parentId: '', sort: '0' })
    }
    setError('')
  }, [editDept, dialogOpen])

  function toggleExpand(id: string) {
    setExpanded(prev => {
      const next = new Set(prev)
      next.has(id) ? next.delete(id) : next.add(id)
      return next
    })
  }

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    setSaving(true)
    setError('')
    try {
      const url = editDept ? `/api/departments/${editDept.id}` : '/api/departments'
      const method = editDept ? 'PATCH' : 'POST'
      const res = await fetch(url, {
        method,
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          name: form.name,
          description: form.description,
          parentId: form.parentId || null,
          sort: Number(form.sort),
        }),
      })
      const data = await res.json()
      if (!res.ok) { setError(data.error); return }
      fetchDepts()
      setDialogOpen(false)
    } catch {
      setError('网络错误')
    } finally {
      setSaving(false)
    }
  }

  async function handleDelete(dept: Department) {
    if (!confirm(`确认删除「${dept.name}」？`)) return
    const res = await fetch(`/api/departments/${dept.id}`, { method: 'DELETE' })
    const data = await res.json()
    if (!res.ok) { alert(data.error); return }
    fetchDepts()
  }

  // 递归渲染部门树
  function renderTree(items: Department[], depth = 0) {
    return items.map(dept => (
      <div key={dept.id}>
        <div
          className={`flex items-center gap-2 py-2 px-3 rounded-lg hover:bg-muted/50 group`}
          style={{ paddingLeft: `${depth * 24 + 12}px` }}
        >
          {/* 展开/收起按钮 */}
          <button
            className="w-5 h-5 flex items-center justify-center text-muted-foreground"
            onClick={() => dept.children?.length && toggleExpand(dept.id)}
          >
            {dept.children?.length > 0
              ? (expanded.has(dept.id) ? <ChevronDown className="h-4 w-4" /> : <ChevronRight className="h-4 w-4" />)
              : <span className="w-4" />
            }
          </button>

          <Building className="h-4 w-4 text-muted-foreground shrink-0" />

          <span className="font-medium flex-1">{dept.name}</span>

          {dept.description && (
            <span className="text-sm text-muted-foreground hidden sm:block">{dept.description}</span>
          )}

          <Badge variant="secondary" className="text-xs">
            {dept._count.users} 人
          </Badge>

          {/* 操作按钮（hover 时显示）*/}
          <div className="flex gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
            <Button
              variant="ghost" size="icon" className="h-7 w-7"
              onClick={() => {
                // 新增子部门时预填 parentId
                setEditDept(null)
                setForm({ name: '', description: '', parentId: dept.id, sort: '0' })
                setDialogOpen(true)
              }}
              title="新增子部门"
            >
              <Plus className="h-3.5 w-3.5" />
            </Button>
            <Button
              variant="ghost" size="icon" className="h-7 w-7"
              onClick={() => { setEditDept(dept); setDialogOpen(true) }}
            >
              <Pencil className="h-3.5 w-3.5" />
            </Button>
            <Button
              variant="ghost" size="icon" className="h-7 w-7 text-destructive hover:text-destructive"
              onClick={() => handleDelete(dept)}
            >
              <Trash2 className="h-3.5 w-3.5" />
            </Button>
          </div>
        </div>

        {/* 展开时递归渲染子部门 */}
        {expanded.has(dept.id) && dept.children?.length > 0 && (
          <div>{renderTree(dept.children, depth + 1)}</div>
        )}
      </div>
    ))
  }

  return (
    <div>
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-bold">部门管理</h1>
        <Button onClick={() => { setEditDept(null); setForm({ name: '', description: '', parentId: '', sort: '0' }); setDialogOpen(true) }}>
          <Plus className="mr-2 h-4 w-4" />
          新增部门
        </Button>
      </div>

      <Card>
        <CardHeader />
        <CardContent>
          {loading ? (
            <div className="flex justify-center py-12">
              <Loader2 className="h-6 w-6 animate-spin text-muted-foreground" />
            </div>
          ) : depts.length === 0 ? (
            <div className="text-center py-12 text-muted-foreground">
              暂无部门数据，点击右上角新增
            </div>
          ) : (
            <div className="space-y-1">
              {renderTree(depts)}
            </div>
          )}
        </CardContent>
      </Card>

      {/* 新增/编辑弹窗 */}
      <Dialog open={dialogOpen} onOpenChange={setDialogOpen}>
        <DialogContent className="max-w-md">
          <DialogHeader>
            <DialogTitle>{editDept ? '编辑部门' : '新增部门'}</DialogTitle>
          </DialogHeader>
          <form onSubmit={handleSubmit} className="space-y-4">
            <div className="space-y-2">
              <Label>部门名称 *</Label>
              <Input
                placeholder="如：技术部"
                value={form.name}
                onChange={e => setForm(f => ({ ...f, name: e.target.value }))}
                required
              />
            </div>

            <div className="space-y-2">
              <Label>描述</Label>
              <Input
                placeholder="部门描述（可选）"
                value={form.description}
                onChange={e => setForm(f => ({ ...f, description: e.target.value }))}
              />
            </div>

            <div className="space-y-2">
              <Label>父级部门</Label>
              <select
                className="w-full border rounded-md px-3 py-2 text-sm bg-background"
                value={form.parentId}
                onChange={e => setForm(f => ({ ...f, parentId: e.target.value }))}
              >
                <option value="">— 顶级部门 —</option>
                {allDepts
                  .filter(d => d.id !== editDept?.id)  // 不能选自己做父级
                  .map(d => (
                    <option key={d.id} value={d.id}>{d.name}</option>
                  ))
                }
              </select>
            </div>

            <div className="space-y-2">
              <Label>排序（数字越小越靠前）</Label>
              <Input
                type="number"
                value={form.sort}
                onChange={e => setForm(f => ({ ...f, sort: e.target.value }))}
              />
            </div>

            {error && <p className="text-sm text-destructive">{error}</p>}

            <DialogFooter>
              <Button type="button" variant="outline" onClick={() => setDialogOpen(false)}>取消</Button>
              <Button type="submit" disabled={saving}>
                {saving && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
                {editDept ? '保存' : '创建'}
              </Button>
            </DialogFooter>
          </form>
        </DialogContent>
      </Dialog>
    </div>
  )
}
```
